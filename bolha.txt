function sort(int[list] lista) -> int[list]
begin
  int bolha, topo, tam, aux;
  tam = len(lista);
  for topo in range(tam,1,-1) do {
     for bolha in range(0,topo-1) do {
        if (lista[bolha] > lista[bolha+1]) then {
           aux = lista[bolha];
           lista[bolha] = lista[bolha+1];
           lista[bolha+1] = aux;
        }
     }
  }
  return lista;
end

function imprime(int[list] lista)
begin
   int i;
   write("\nLista ordenada: \n");
   for i in lista do {
      write(i, "\n");
   }
   write("FIM\n");
end

function main()
begin
   string prompt, entrada;
   int n, aux;
   int[list] lista;
   write("Forneca inteiros para ordenar (0=termina):");
   n = 1;
   prompt = num2str(n) + "o: ";
   lista = [];
   while (aux != 0) {
      read("msg: ", entrada);
      aux = str2num(entrada);
      lista = lista + [aux];
   }
   lista = sort(lista);
   imprime(lista);
   return;
end
